# StudentLearn Fullstack: Easy Setup Guide

## 1. Clone the repository (if you haven't already)
git clone <your-repo-url>
cd <repo-root>

## 2. BACKEND SETUP
cd backend

# (Optional, but recommended) Create a Python virtual environment:
python3 -m venv venv
source venv/bin/activate

# Install backend dependencies:
pip install --break-system-packages -r requirements.txt

# Copy and edit the environment file:
cp env.example .env
# Edit .env to set your secrets and config (nano .env or use your editor)

# Generate the database (SQLite by default):
python3 -c "from app.database import engine; from app.models import Base; Base.metadata.create_all(bind=engine)"

# (Recommended) Create an admin user:
python3 create_admin.py

# Start the backend server:
uvicorn app.main:app --reload --host 0.0.0.0 --port 8000

# (Leave this running, or open a new terminal for the frontend)

## 3. FRONTEND SETUP
cd ../

# Install frontend dependencies:
npm install

# Copy and edit the environment file if needed:
cp backend/env.example .env.local
# Edit .env.local to set NEXT_PUBLIC_API_URL and other frontend config

# Start the frontend server:
npm run dev

# The frontend will be available at:
http://localhost:3000

# The backend API docs will be available at:
http://localhost:8000/docs

---

# For production:
- Use a real database (PostgreSQL/MySQL) and update DATABASE_URL in .env
- Set strong secrets in .env/.env.local
- Use a process manager (gunicorn, pm2, etc.)
- Serve frontend with a production build (npm run build && npm start)
